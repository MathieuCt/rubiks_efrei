\doxysubsection{Description détaillée}
Définition des cubies, qui sont les petits cubes de couleur rattachées à une face 
\begin{DoxyParams}{Paramètres}
{\em x} & \+: coordonnée horizontale du cubie \\
\hline
{\em y} & \+: coordonnée verticale du cubie \\
\hline
{\em num} & \+: Numéro du cubie dans la face \+: de 0 à 8 \\
\hline
{\em type} & \+: Type de cubie, CORNER, EDGE ou CENTER \\
\hline
{\em color} & \+: Couleur du cubie \\
\hline
{\em cubie\+\_\+side} & \+: Face de rattachement du cubie \\
\hline
{\em neighbours\mbox{[}2\mbox{]}} & \+: Définition des faces adjacentes au cubie. Il peut y en avoir 2 pour un CORNER et 1 pour un EDGE \\
\hline
\end{DoxyParams}
